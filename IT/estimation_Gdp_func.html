Estimation of GDP Function

\beg in{array}{l} \frac{\partial \ln y_{i}}{\partial \ln V_{k}}=\frac{\partial \ln \left(s_{i} G / p_{i}\right)}{\partial \ln V_{k}}=\frac{\partial \ln s_{i}}{\partial \ln V_{k}}+\frac{\partial \ln G}{\partial \ln V_{k}}-\frac{\partial \ln p_{i}}{\partial \ln V_{k}} \\ =\left(\frac{\partial s_{i}}{\partial V_{k}}\right)\left(\frac{V_{k}}{s_{i}}\right)+\left(\frac{\partial G}{\partial V_{k}}\right)\left(\frac{V_{k}}{G}\right) \\ =\left(\frac{\phi_{i k}}{V_{k}}\right)\left(\frac{V_{k}}{s_{i}}\right)+\frac{w V_{k }}{G} \\ =\frac{\phi_{i k}}{s_{i}}+s_{k} \end{array}
